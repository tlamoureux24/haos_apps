#include <tunables/global>

profile rsync_manager flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/bash>
  #include <abstractions/nameservice>

  # Base container and service permissions
  # mount.cifs adjusts this exact capability set before mounting.
  capability dac_override,
  capability dac_read_search,
  capability setpcap,
  capability sys_admin,
  capability setgid,
  capability setuid,
  file,
  network,
  signal,
  mount,
  umount,

  # S6-overlay startup and services
  /init ix,
  /bin/** ix,
  /sbin/** ix,
  /usr/bin/** ix,
  /usr/sbin/** ix,
  /usr/local/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio and temporary files
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # App persistent data and generated crontab
  /data/** rw,
  /var/spool/cron/crontabs/** rwk,

  # Ingress web UI and CGI scripts
  /www/** rix,
  /etc/lighttpd/** r,

  # Home Assistant folders mapped by config.yaml
  /share/** rw,
  /media/** rw,
  /backup/** rw,

  # Rsync Manager network mounts
  /mnt/** rwk,

  # Logging through the container init stdout/stderr descriptors.
  /proc/1/fd/1 rw,
  /proc/1/fd/2 rw,
  /proc/[0-9]*/fd/** rw,

  # Process discovery used by pidof/crond/service supervision.
  /proc/** r,
}
